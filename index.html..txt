    <script>
        const apiKey = ""; // API Key provided at runtime
        let lastAIResponseText = "";

        // Helper: Exponential Backoff Fetch
        async function fetchWithRetry(url, options, maxRetries = 5) {
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (response.ok) return await response.json();
                    if (response.status === 429) throw new Error('Rate limit');
                    throw new Error(HTTP error! status: ${response.status});
                } catch (err) {
                    if (i === maxRetries - 1) throw err;
                    await new Promise(res => setTimeout(res, Math.pow(2, i) * 1000));
                }
            }
        }

        // --- Feature 1: AI Consultant (Text & TTS) ---
        async function askGemini() {
            const input = document.getElementById('aiInput').value.trim();
            const responseContainer = document.getElementById('aiResponseContainer');
            const responseText = document.getElementById('aiText');
            const loadingIndicator = document.getElementById('aiLoading');
            const ttsControls = document.getElementById('ttsControls');

            if (!input) return;

            responseContainer.classList.remove('hidden');
            loadingIndicator.classList.remove('hidden');
            ttsControls.classList.add('hidden');
            responseText.innerText = "";

            const systemPrompt = "أنت مساعد ذكي لمدرسة الرازي الإعدادية (جلجولية). أجب بأسلوب ملهم ومهني. شجع الأهالي على التسجيل للعام 2026-2027. استعمل دائماً جملة 'رازي وأفتخر' في الختام.";
            
            try {
                const url = https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey};
                const result = await fetchWithRetry(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: input }] }],
                        systemInstruction: { parts: [{ text: systemPrompt }] }
                    })
                });

                const text = result.candidates?.[0]?.content?.parts?.[0]?.text || "نعتذر، لم أتمكن من العثور على إجابة.";
                responseText.innerText = text;
                lastAIResponseText = text;
                ttsControls.classList.remove('hidden');
            } catch (error) {
                responseText.innerText = "عذراً، حدث خطأ. يرجى المحاولة لاحقاً.";
            } finally {
                loadingIndicator.classList.add('hidden');
            }
        }

        async function playAIResponse() {
            if (!lastAIResponseText) return;
            const url = https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key=${apiKey};
            
            try {
                const result = await fetchWithRetry(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: بصوت ملهم: ${lastAIResponseText} }] }],
                        generationConfig: {
                            responseModalities: ["AUDIO"],
                            speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Kore" } } }
                        }
                    })
                });

                const audioBase64 = result.candidates?.[0]?.content?.parts?.[0]?.inlineData?.data;
                const mimeType = result.candidates?.[0]?.content?.parts?.[0]?.inlineData?.mimeType;
                
                if (audioBase64) {
                    const sampleRate = parseInt(mimeType.match(/rate=(\d+)/)?.[1] || "24000");
                    const audioBlob = pcmToWav(audioBase64, sampleRate);
                    const audioUrl = URL.createObjectURL(audioBlob);
                    const audio = new Audio(audioUrl);
                    audio.play();
                }
            } catch (err) {
                console.error("TTS Error:", err);
            }
        }

        function pcmToWav(base64Pcm, sampleRate) {
            const binaryString = atob(base64Pcm);
            const len = binaryString.length;
            const buffer = new ArrayBuffer(44 + len);
            const view = new DataView(buffer);

            const writeString = (offset, string) => {
                for (let i = 0; i < string.length; i++) {
                    view.setUint8(offset + i, string.charCodeAt(i));
                }
            };

            writeString(0, 'RIFF');
            view.setUint32(4, 36 + len, true);
            writeString(8, 'WAVE');
            writeString(12, 'fmt ');
            view.setUint32(16, 16, true);
            view.setUint16(20, 1, true);
            view.setUint16(22, 1, true);
            view.setUint32(24, sampleRate, true);
            view.setUint32(28, sampleRate * 2, true);
            view.setUint16(32, 2, true);
            view.setUint16(34, 16, true);
            writeString(36, 'data');
            view.setUint32(40, len, true);

            for (let i = 0; i < len; i++) {
                view.setUint8(44 + i, binaryString.charCodeAt(i));
            }
            return new Blob([buffer], { type: 'audio/wav' });
        }

        // --- Feature 2: Student Card Generator (Imagen 4.0) ---
        async function generateStudentCard() {
            const name = document.getElementById('studentName').value.trim();
            const hobby = document.getElementById('studentHobby').value.trim();
            const container = document.getElementById('imageContainer');
            const imgEl = document.getElementById('generatedImage');
            const loading = document.getElementById('imageLoading');
            const status = document.getElementById('imageStatus');

            if (!name || !hobby) {
                alert("يرجى إدخال اسم الطالب وهوايته.");
                return;
            }

            container.classList.remove('hidden');
            loading.classList.remove('hidden');
            imgEl.classList.add('hidden');
            status.innerText = "جاري تصميم بطاقة الطالب المبدعة...";

            const prompt = A cinematic, high-quality 3D digital illustration of a student named ${name} who loves ${hobby}. The student is wearing a school uniform with a blue badge, standing heroically in a modern, high-tech school hallway of "Al-Razi School". The atmosphere is inspiring and futuristic. No text on image except possibly school logo vibes.;

            try {
                const url = https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey};
                const result = await fetchWithRetry(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        instances: [{ prompt: prompt }],
                        parameters: { sampleCount: 1 }
                    })
                });

                const base64 = result.predictions?.[0]?.bytesBase64Encoded;
                if (base64) {
                    imgEl.src = data:image/png;base64,${base64};
                    imgEl.classList.remove('hidden');
                    status.innerText = بطاقة الطالب ${name} جاهزة! ✨;
                } else {
                    throw new Error("No image generated");
                }
            } catch (error) {
                status.innerText = "عذراً، لم نتمكن من توليد الصورة حالياً.";
            } finally {
                loading.classList.add('hidden');
            }
        }
